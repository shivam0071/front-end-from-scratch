  • Splicing an Array 
    
    // Splicing an Array -- Inplace, modifies original list 
    // splice can be used to remove, add, or replace elements in an array.
    
    // array.splice(start, deleteCount, item1, item2, ...)
    // deleteCount: The number of elements to remove from the array.

    let fruits = ['Apple', 'Banana', 'Cherry', 'Date'];
    let removed = fruits.splice(1, 2); // Start at index 1, remove 2 elements
    console.log(fruits); // Output: ['Apple', 'Date']
    console.log(removed); // Output: ['Banana', 'Cherry']

    let numbers = [1, 2, 3];
    numbers.splice(1, 0, 4, 5); // Start at index 1, remove 0 elements, add 4 and 5
    console.log(numbers); // Output: [1, 4, 5, 2, 3]

    let colors = ['Red', 'Green', 'Blue'];
    colors.splice(1, 1, 'Yellow'); // Start at index 1, remove 1 element, add 'Yellow'
    console.log(colors); // Output: ['Red', 'Yellow', 'Blue']

    let animals = ['Dog', 'Cat', 'Fish', 'Bird'];
    animals.splice(2); // Start at index 2, remove all elements from that index
    console.log(animals); // Output: ['Dog', 'Cat']
